import { ALBEvent, ALBResult } from 'aws-lambda';

export const handler = async (event: ALBEvent): Promise<ALBResult> => {
    const name = event.queryStringParameters?.name || 'World';
    return {
        statusCode: 200,
        statusDescription: '200 OK',
        isBase64Encoded: false,
        body: `Hello, ${name}!`,
    };
};
